## RV32 Emulator Testing Assembly Code

        li t0, 5          
        li t1, 0          
        
        beq zero, zero, branch1  
        addi t1, t1, 100   
branch1:
        addi t1, t1, 1     
        
        bne t0, t0, branch2  
        addi t1, t1, 1     
branch2:
        
        jal zero,  jump1
        addi t1, t1, 100   
jump1:
        addi t1, t1, 1     
        
        jal func1           
        jal zero, end               
        
func1:
        addi t1, t1, 1     
        jalr zero, 0(ra)              
        
end:
        HCF
